distortions=[]

for i in K_range:
    start_time = time.time()
    kmeanModel = KMeans(n_clusters=i,init='k-means++')
    kmeanModel.fit(X)
    distortions.append(sum(np.min(cdist(X, kmeanModel.cluster_centers_, 'euclidean'),axis=1)) / X.shape[0])
